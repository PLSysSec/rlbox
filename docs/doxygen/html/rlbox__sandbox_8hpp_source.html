114 // This variable tracks of the sandbox has already been created/destroyed.
115 // APIs in this class should be called only when the sandbox is created.
116 // However, it is expensive to check in APIs such as invoke or in the callback
117 // interceptor. What's more, there could be time of check time of use issues
118 // in the checks as well.
119 // In general, we leave it up to the user to ensure these APIs are never

141 template<typename T>
142 inline constexpr void check_invoke_param_type_is_ok()
143 {
144 using T_NoRef = std::remove_reference_t<T>;
145
146 if_constexpr_named(cond1, detail::rlbox_is_wrapper_v<T_NoRef>)
147 {
148 if_constexpr_named(
149 subcond1,
150 !std::is_same_v<T_Sbx, detail::rlbox_get_wrapper_sandbox_t<T_NoRef>>)
151 {
152 rlbox_detail_static_fail_because(
153 cond1 && subcond1,
154 "Mixing tainted data from a different sandbox types. This could "
155 "happen due to couple of different reasons.\n"
156 "1. You are using 2 sandbox types for example'rlbox_noop_sandbox' "
157 "and 'rlbox_lucet_sandbox', and are passing tainted data from one "
158 "sandbox as parameters into a function call to the other sandbox. "
159 "This is not allowed, unwrap the tainted data with copy_and_verify "
160 "or other unwrapping APIs first.\n"
161 "2. You have inadvertantly forgotten to set/remove "
162 "RLBOX_USE_STATIC_CALLS depending on the sandbox type. Some sandbox "
163 "types like rlbox_noop_sandbox require this to be set to a given "
164 "value, while other types like rlbox_lucet_sandbox, require this not "
165 "to be set.");
166 }
167 }
168 else if_constexpr_named(cond2,
169 std::is_null_pointer_v<T_NoRef> ||
170 detail::is_fundamental_or_enum_v<T_NoRef>)
171 {}
172 else
173 {
174 constexpr auto unknownCase = !(cond1 || cond2);
175 rlbox_detail_static_fail_because(
176 unknownCase,
177 "Arguments to a sandbox function call should be primitives or wrapped "
178 "types like tainted, callbacks etc.");
179 }
180 }

182 template<typename T>
183 inline auto invoke_process_param(T&& param)
184 {
185 check_invoke_param_type_is_ok<T>();
186
187 using T_NoRef = std::remove_reference_t<T>;
188
189 if constexpr (detail::rlbox_is_tainted_opaque_v<T_NoRef>) {
190 auto ret = from_opaque(param);
191 return ret.UNSAFE_sandboxed(*this);
192 } else if constexpr (detail::rlbox_is_wrapper_v<T_NoRef>) {
193 return param.UNSAFE_sandboxed(*this);
194 } else if constexpr (std::is_null_pointer_v<T_NoRef>) {
195 tainted<void*, T_Sbx> ret = nullptr;
196 return ret.UNSAFE_sandboxed(*this);
197 } else if constexpr (detail::is_fundamental_or_enum_v<T_NoRef>) {
198 // For unwrapped primitives, assign to a tainted var and then unwrap so
199 // that we adjust for machine model
200 tainted<T_NoRef, T_Sbx> ret = param;
201 return ret.UNSAFE_sandboxed(*this);
202 } else {
203 rlbox_detail_static_fail_because(detail::true_v<T_NoRef>,
204 "Only tainted types, callbacks or primitive values such as ints can be passed as parameters.\n"
205 "To make a parameter tainted, try moving the allocation into the sandbox.\n"
206 "If the parameter is a callback, try registering the callback via the register_callback API.");
207 }
208 }

227 template<typename T_Ret, typename... T_Args>
228 static detail::convert_to_sandbox_equivalent_t<T_Ret, T_Sbx>
229 sandbox_callback_interceptor(
230 detail::convert_to_sandbox_equivalent_t<T_Args, T_Sbx>... args)
231 {
232 std::pair<T_Sbx*, void*> context =
233 T_Sbx::impl_get_executed_callback_sandbox_and_key();
234 auto& sandbox = *(reinterpret_cast<rlbox_sandbox<T_Sbx>*>(context.first));
235 auto key = context.second;
236
237 using T_Func_Ret =
238 std::conditional_t<std::is_void_v<T_Ret>, void, tainted<T_Ret, T_Sbx>>;
239 using T_Func =
240 T_Func_Ret (*)(rlbox_sandbox<T_Sbx>&, tainted<T_Args, T_Sbx>...);
241 auto target_fn_ptr = reinterpret_cast<T_Func>(key);
242
243#ifdef RLBOX_MEASURE_TRANSITION_TIMES
244 high_resolution_clock::time_point enter_time = high_resolution_clock::now();
245 auto on_exit = rlbox::detail::make_scope_exit([&] {
246 auto exit_time = high_resolution_clock::now();
247 int64_t ns = duration_cast<nanoseconds>(exit_time - enter_time).count();
248 sandbox.transition_times.push_back(
249 rlbox_transition_timing{ rlbox_transition::CALLBACK,
250 nullptr /* func_name */,
251 key /* func_ptr */,
252 ns });
253 });
254#endif
255#ifdef RLBOX_TRANSITION_ACTION_OUT
256 RLBOX_TRANSITION_ACTION_OUT(rlbox_transition::CALLBACK,
257 nullptr /* func_name */,
258 key /* func_ptr */,
259 sandbox.transition_state);
260#endif
261#ifdef RLBOX_TRANSITION_ACTION_IN
262 auto on_exit_transition = rlbox::detail::make_scope_exit([&] {
263 RLBOX_TRANSITION_ACTION_IN(rlbox_transition::CALLBACK,
264 nullptr /* func_name */,
265 key /* func_ptr */,
266 sandbox.transition_state);
267 });
268#endif
269 if constexpr (std::is_void_v<T_Func_Ret>) {
270 (*target_fn_ptr)(
271 sandbox,
272 sandbox.template sandbox_callback_intercept_convert_param<T_Args>(
273 sandbox, args)...);
274 return;
275 } else {
276 auto tainted_ret = (*target_fn_ptr)(
277 sandbox,
278 sandbox.template sandbox_callback_intercept_convert_param<T_Args>(
279 sandbox, args)...);
280
281 using namespace detail;
282 convert_to_sandbox_equivalent_t<T_Ret, T_Sbx> ret;
283 convert_type<T_Sbx,
284 adjust_type_direction::TO_SANDBOX,
285 adjust_type_context::SANDBOX>(
286 ret,
287 tainted_ret.get_raw_value_ref(),
288 nullptr /* example_unsandboxed_ptr */,
289 &sandbox);
290 return ret;
291 }
292 }

298 template<typename T_Ret, typename... T_Args>
299 inline void unregister_callback(void* key)
300 {
301 // Silently swallowing the failure is better here as RAII types may try to
302 // cleanup callbacks after sandbox destruction
303 if (sandbox_created.load() != Sandbox_Status::CREATED) {
304 return;
305 }
306
307 this->template impl_unregister_callback<
308 detail::convert_to_sandbox_equivalent_t<T_Ret, T_Sbx>,
309 detail::convert_to_sandbox_equivalent_t<T_Args, T_Sbx>...>(key);
310
311 std::lock_guard<std::mutex> lock(callback_lock);
312 auto el_ref = std::find(callback_keys.begin(), callback_keys.end(), key);
313 detail::dynamic_check(
314 el_ref != callback_keys.end(),
315 "Unexpected state. Unregistering a callback that was never registered.");
316 callback_keys.erase(el_ref);
317 }

368 template<typename... T_Args>
369 inline bool create_sandbox(T_Args... args)
370 {
371#ifdef RLBOX_MEASURE_TRANSITION_TIMES
372 // Warm up the timer. The first call is always slow (at least on the test
373 // platform)
374 for (int i = 0; i < 10; i++) {
375 auto val = high_resolution_clock::now();
376 RLBOX_UNUSED(val);
377 }
378#endif
379 auto expected = Sandbox_Status::NOT_CREATED;
380 bool success = sandbox_created.compare_exchange_strong(
381 expected, Sandbox_Status::INITIALIZING /* desired */);
382 detail::dynamic_check(
383 success,
384 "create_sandbox called when sandbox already created/is being "
385 "created concurrently");
386
387 using T_Result = rlbox::detail::polyfill::invoke_result_t<decltype(impl_create_sandbox_helper<T_Args...>), decltype(this), T_Args...>;
388
389 bool created = true;
390 if constexpr (std::is_same_v<T_Result, void>) {
391 this->impl_create_sandbox(std::forward<T_Args>(args)...);
392 } else if constexpr (std::is_same_v<T_Result, bool>) {
393 created = this->impl_create_sandbox(std::forward<T_Args>(args)...);
394 } else {
395 rlbox_detail_static_fail_because(
396 (!std::is_same_v<T_Result, void> && !std::is_same_v<T_Result, bool>),
397 "Expected impl_create_sandbox to return void or a boolean"
398 );
399 }
400
401 if (created) {
402 sandbox_created.store(Sandbox_Status::CREATED);
403 RLBOX_ACQUIRE_UNIQUE_GUARD(lock, sandbox_list_lock);
404 sandbox_list.push_back(this);
405 }
406
407 return created;
408 }

413 inline auto destroy_sandbox()
414 {
415 auto expected = Sandbox_Status::CREATED;
416 bool success = sandbox_created.compare_exchange_strong(
417 expected, Sandbox_Status::CLEANING_UP /* desired */);
418
419 detail::dynamic_check(
420 success,
421 "destroy_sandbox called without sandbox creation/is being "
422 "destroyed concurrently");
423
424 {
425 RLBOX_ACQUIRE_UNIQUE_GUARD(lock, sandbox_list_lock);
426 auto el_ref = std::find(sandbox_list.begin(), sandbox_list.end(), this);
427 detail::dynamic_check(
428 el_ref != sandbox_list.end(),
429 "Unexpected state. Destroying a sandbox that was never initialized.");
430 sandbox_list.erase(el_ref);
431 }
432
433 sandbox_created.store(Sandbox_Status::NOT_CREATED);
434 return this->impl_destroy_sandbox();
435 }

516 template<typename T>
517 inline tainted<T*, T_Sbx> malloc_in_sandbox(uint32_t count)
518 {
519 // Silently swallowing the failure is better here as RAII types may try to
520 // malloc after sandbox destruction
521 if (sandbox_created.load() != Sandbox_Status::CREATED) {
522 return tainted<T*, T_Sbx>::internal_factory(nullptr);
523 }
524
525 detail::dynamic_check(count != 0, "Malloc tried to allocate 0 bytes");
526 if constexpr (sizeof(T) >= std::numeric_limits<uint32_t>::max()) {
527 rlbox_detail_static_fail_because(sizeof(T) >=
528 std::numeric_limits<uint32_t>::max(),
529 "Tried to allocate an object over 4GB.");
530 }
531 auto total_size = static_cast<uint64_t>(sizeof(T)) * count;
532 if constexpr (sizeof(size_t) == 4) {
533 // On a 32-bit platform, we need to make sure that total_size is not >=4GB
534 detail::dynamic_check(total_size < std::numeric_limits<uint32_t>::max(),
535 "Tried to allocate memory over 4GB");
536 } else if constexpr (sizeof(size_t) != 8) {
537 // Double check we are on a 64-bit platform
538 // Note for static checks we need to have some dependence on T, so adding
539 // a dummy
540 constexpr bool dummy = sizeof(T) >= 0;
541 rlbox_detail_static_fail_because(dummy && sizeof(size_t) != 8,
542 "Expected 32 or 64 bit platform.");
543 }
544 auto ptr_in_sandbox = this->impl_malloc_in_sandbox(total_size);
545 auto ptr = get_unsandboxed_pointer<T*>(ptr_in_sandbox);
546 if (!ptr) {
547 return tainted<T*, T_Sbx>(nullptr);
548 }
549 detail::dynamic_check(is_pointer_in_sandbox_memory(ptr),
550 "Malloc returned pointer outside the sandbox memory");
551 auto ptr_end = reinterpret_cast<uintptr_t>(ptr + (count - 1));
552 detail::dynamic_check(
553 is_in_same_sandbox(ptr, reinterpret_cast<void*>(ptr_end)),
554 "Malloc returned a pointer whose range goes beyond sandbox memory");
555 auto cast_ptr = reinterpret_cast<T*>(ptr);
556 return tainted<T*, T_Sbx>::internal_factory(cast_ptr);
557 }

729 // this is an internal function invoked from macros, so it has be public
730 // Explicitly don't use inline on this, as this adds a lot of instructions
731 // prior to function call. What's more, by not inlining, different function
732 // calls with the same signature can share the same code segments for
733 // sandboxed function execution in the binary
734 template<typename T, typename... T_Args>
735 auto INTERNAL_invoke_with_func_ptr(const char* func_name,
736 void* func_ptr,
737 T_Args&&... params)
738 {
739 // unused in some paths
740 RLBOX_UNUSED(func_name);
741#ifdef RLBOX_MEASURE_TRANSITION_TIMES
742 auto enter_time = high_resolution_clock::now();

841 template<typename T_RL, typename T_Ret, typename... T_Args>
842 sandbox_callback<T_Cb_no_wrap<T_Ret, T_Args...>*, T_Sbx> register_callback(
843 T_Ret (*func_ptr)(T_RL, T_Args...))
844 {
845 // Some branches don't use the param
846 RLBOX_UNUSED(func_ptr);
847
848 if_constexpr_named(cond1, !std::is_same_v<T_RL, rlbox_sandbox<T_Sbx>&>)
849 {
850 rlbox_detail_static_fail_because(
851 cond1,
852 "Modify the callback to change the first parameter to a sandbox. "
853 "For instance if a callback has type\n\n"
854 "int foo(int a, int b) {...}\n\n"
855 "Change this to \n\n"
856 "tainted<int, T_Sbx> foo(rlbox_sandbox<T_Sbx>& sandbox, "
857 "tainted<int, T_Sbx> a, tainted<int, T_Sbx> b) {...}\n");
858 }
859 else if_constexpr_named(
860 cond2, !(detail::rlbox_is_tainted_or_opaque_v<T_Args> && ...))
861 {
862 rlbox_detail_static_fail_because(
863 cond2,
864 "Change all arguments to the callback have to be tainted or "
865 "tainted_opaque. "
866 "For instance if a callback has type\n\n"
867 "int foo(int a, int b) {...}\n\n"
868 "Change this to \n\n"
869 "tainted<int, T_Sbx> foo(rlbox_sandbox<T_Sbx>& sandbox, "
870 "tainted<int, T_Sbx> a, tainted<int, T_Sbx> b) {...}\n");
871 }
872 else if_constexpr_named(
873 cond3, (std::is_array_v<detail::rlbox_remove_wrapper_t<T_Args>> || ...))
874 {
875 rlbox_detail_static_fail_because(
876 cond3,
877 "Change all static array arguments to the callback to be pointers. "
878 "For instance if a callback has type\n\n"
879 "int foo(int a[4]) {...}\n\n"
880 "Change this to \n\n"
881 "tainted<int, T_Sbx> foo(rlbox_sandbox<T_Sbx>& sandbox, "
882 "tainted<int*, T_Sbx> a) {...}\n");
883 }
884 else if_constexpr_named(
885 cond4,
886 !(std::is_void_v<T_Ret> || detail::rlbox_is_tainted_or_opaque_v<T_Ret>))
887 {
888 rlbox_detail_static_fail_because(
889 cond4,
890 "Change the callback return type to be tainted or tainted_opaque if it "
891 "is not void. "
892 "For instance if a callback has type\n\n"
893 "int foo(int a, int b) {...}\n\n"
894 "Change this to \n\n"
895 "tainted<int, T_Sbx> foo(rlbox_sandbox<T_Sbx>& sandbox, "
896 "tainted<int, T_Sbx> a, tainted<int, T_Sbx> b) {...}\n");
897 }
898 else
899 {
900 detail::dynamic_check(
901 sandbox_created.load() == Sandbox_Status::CREATED,
902 "register_callback called without sandbox creation");
903
904 // Need unique key for each callback we register - just use the func addr
905 void* unique_key = reinterpret_cast<void*>(func_ptr);
906
907 // Make sure that the user hasn't previously registered this function...
908 // If they have, we would returning 2 owning types (sandbox_callback) to
909 // the same callback which would be bad
910 {
911 std::lock_guard<std::mutex> lock(callback_lock);
912 bool exists =
913 std::find(callback_keys.begin(), callback_keys.end(), unique_key) !=
914 callback_keys.end();
915 detail::dynamic_check(
916 !exists, "You have previously already registered this callback.");
917 callback_keys.push_back(unique_key);
918 }
919
920 auto callback_interceptor =
921 sandbox_callback_interceptor<detail::rlbox_remove_wrapper_t<T_Ret>,
922 detail::rlbox_remove_wrapper_t<T_Args>...>;
923
924 auto callback_trampoline = this->template impl_register_callback<
925 detail::convert_to_sandbox_equivalent_t<
926 detail::rlbox_remove_wrapper_t<T_Ret>,
927 T_Sbx>,
928 detail::convert_to_sandbox_equivalent_t<
929 detail::rlbox_remove_wrapper_t<T_Args>,
930 T_Sbx>...>(unique_key, reinterpret_cast<void*>(callback_interceptor));
931
932 auto tainted_func_ptr = reinterpret_cast<
933 detail::rlbox_tainted_opaque_to_tainted_t<T_Ret, T_Sbx> (*)(
934 T_RL, detail::rlbox_tainted_opaque_to_tainted_t<T_Args, T_Sbx>...)>(
935 reinterpret_cast<void*>(func_ptr));
936
937 auto ret = sandbox_callback<T_Cb_no_wrap<T_Ret, T_Args...>*, T_Sbx>(
938 this,
939 tainted_func_ptr,
940 callback_interceptor,
941 callback_trampoline,
942 unique_key);
943 return ret;
944 }
945 }
