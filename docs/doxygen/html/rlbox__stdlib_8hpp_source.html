110 T_Wrap<T_Rhs*, T_Sbx> ptr,
111 T_Val value,
112 T_Num num)
113{
114
115 static_assert(detail::rlbox_is_tainted_or_vol_v<T_Wrap<T_Rhs, T_Sbx>>,
116 "memset called on non wrapped type");
117
118 static_assert(!std::is_const_v<T_Rhs>, "Destination is const");
119

121 detail::dynamic_check(num_val <= sandbox.get_total_memory(),

127

221template<typename T_Sbx, typename T>
222tainted<T*, T_Sbx> copy_memory_or_grant_access(rlbox_sandbox<T_Sbx>& sandbox,
223 T* src,
224 size_t num,
225 bool free_source_on_copy,
226 bool& copied)
227{
228 // Malloc in sandbox takes a uint32_t as the parameter, need a bounds check
229 detail::dynamic_check(num <= std::numeric_limits<uint32_t>::max(),
230 "Granting access too large a region");
231 uint32_t num_trunc = num;
232
233 // sandbox can grant access if it includes the following line
234 // using can_grant_deny_access = void;
235 if constexpr (detail::has_member_using_can_grant_deny_access_v<T_Sbx>) {
236 detail::check_range_doesnt_cross_app_sbx_boundary<T_Sbx>(src, num_trunc);
237
238 bool success;
239 auto ret = sandbox.INTERNAL_grant_access(src, num_trunc, success);
240 if (success) {
241 copied = false;
242 return ret;
243 }
244 }
245
246 using T_nocv = std::remove_cv_t<T>;
247 tainted<T_nocv*, T_Sbx> copy =
248 sandbox.template malloc_in_sandbox<T_nocv>(num_trunc);
249 rlbox::memcpy(sandbox, copy, src, num_trunc);
250 if (free_source_on_copy) {
251 free(const_cast<void*>(reinterpret_cast<const void*>(src)));
252 }
253
254 copied = true;
255 return sandbox_const_cast<T*>(copy);
256}

272template<typename T_Sbx,
273 typename T,
274 template<typename, typename>
275 typename T_Wrap>
276T* copy_memory_or_deny_access(rlbox_sandbox<T_Sbx>& sandbox,
277 T_Wrap<T*, T_Sbx> src,
278 size_t num,
279 bool free_source_on_copy,
280 bool& copied)
281{
282 // sandbox can grant access if it includes the following line
283 // using can_grant_deny_access = void;
284 if constexpr (detail::has_member_using_can_grant_deny_access_v<T_Sbx>) {
285 detail::check_range_doesnt_cross_app_sbx_boundary<T_Sbx>(
286 src.INTERNAL_unverified_safe(), num);
287
288 bool success;
289 auto ret = sandbox.INTERNAL_deny_access(src, num, success);
290 if (success) {
291 copied = false;
292 return ret;
293 }
294 }
295
296 auto copy = static_cast<T*>(malloc(num));
297
298 tainted<T*, T_Sbx> src_tainted = src;
299 char* src_raw = src_tainted.copy_and_verify_buffer_address(
300 [](uintptr_t val) { return reinterpret_cast<char*>(val); }, num);
301 std::memcpy(copy, src_raw, num);
302 if (free_source_on_copy) {
303 sandbox.free_in_sandbox(src);
304 }
305
306 copied = true;
307 return copy;
308}
