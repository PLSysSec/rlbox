187 // Scope Exit guards
188 template<typename T_ExitFunc>
189 class scope_exit
190 {
191 T_ExitFunc exit_func;
192 bool released;
193
194 public:
195 explicit scope_exit(T_ExitFunc&& cleanup)
196 : exit_func(cleanup)
197 , released(true)
198 {}
199
200 scope_exit(scope_exit&& rhs)
201 : exit_func(std::move(rhs.exit_func))
202 , released(rhs.released)
203 {
204 rhs.release();
205 }
206
207 ~scope_exit()
208 {
209 if (released) {
210 exit_func();
211 }
212 }
213
214 void release() { released = false; }
215
216 private:
217 explicit scope_exit(const scope_exit&) = delete;
218 scope_exit& operator=(const scope_exit&) = delete;
219 scope_exit& operator=(scope_exit&&) = delete;
220 };

222 template<typename T_ExitFunc>
223 [[nodiscard]] scope_exit<T_ExitFunc> make_scope_exit(
224 T_ExitFunc&& exitFunction)
